%%% §1
\sechead{1. 背景と目的}
\begin{itemize}
  \item 鏡面反射物体は光沢・映り込みにより外観が不安定 → 教師なし手法に深刻な課題
  \item SAVi-DINOSAUR: 凍結ViT特徴量 + Slot Attention で教師なしセグメンテーション
  \item 3バックボーン (DINOv2 / DINOv1 / CLIP) と金属・ゴム 2素材で比較評価
  \item FG-ARI・mBO・Slot数 Kの感度、τ閾値の影響を系統的に分析
\end{itemize}
\vspace{1mm}
\colorbox{lightyellow}{\parbox{\dimexpr\linewidth-2\fboxsep}{%
  \textbf{課題設定:} 凍結ViT特徴量の質がセグメンテーション性能を左右するか？
  金属光沢面に対して各バックボーンはどのような失敗モードを示すか？
}}\vspace{3mm}

%%% §4
\sechead{4. DINOv2 高FG-ARIの要因}
\begin{center}
  \includegraphics[width=\linewidth,height=80mm,keepaspectratio]{figures/paper_3backbone_k11_comparison.png}
\end{center}
\figcap{図2: 3バックボーン $K{=}11$ 比較（金属シーン）。上: DINOv2、中: DINOv1、下: CLIP}
\vspace{1mm}
\colorbox{lightyellow}{\parbox{\dimexpr\linewidth-2\fboxsep}{%
  \textbf{Goldilocks 特性:} $K{=}9$ 付近で FG-ARI が最大化後に飽和する。
  スロット数が多すぎると物体が分割され、少なすぎると背景と混合する。
}}\vspace{2mm}
\begin{itemize}
  \item DINOv2: スロットが物体境界に整合、金属面も局在化成功
  \item DINOv1: チャンネル正規化後も背景との分離が不安定
  \item CLIP: 言語的意味論が優先 → テクスチャ境界を無視し崩壊
\end{itemize}

%%% §9
\sechead{9. まとめと今後の課題}
\begin{itemize}
  \item DINOv2 が金属シーンで DINOv1・CLIP を大幅に上回ることを定量的に確認
  \item Goldilocks 特性: $K{=}9$ が最適、過少・過多でともに性能低下
  \item τ感度: $\tau{=}0.05$ が最適、頑健性は中程度
  \item エントロピーと時系列追跡で失敗モードを定性的に可視化
\end{itemize}
\vspace{2mm}
\begin{itemize}
  \item 鏡面反射に特化した損失関数の設計が必要
  \item DINOv1・CLIP でのスロット崩壊の根本原因の解明
  \item 実物体に対するゼロショット汎化性能の検証
\end{itemize}
\vspace{2mm}
\colorbox{lightyellow}{\parbox{\dimexpr\linewidth-2\fboxsep}{%
  \textbf{Limitations:} 本研究は MOVi-A の合成データのみを対象としており、
  実世界の鏡面物体への汎化は未検証。チェックポイントが公開されていないため
  完全な再現実験には独自学習が必要。
}}
